#include<stdio.h>
int main()
{
        int a[100],i,j,k,n,t;
        
        printf("ENTER LIMIT");
        scanf("%d",&n);
        printf("enter %d number",n);
        for(i=0;i<n;i++)
        {
                scanf("%d",&a[i]);
        }
        
        for(i=0;i<n;i++)
        {
                for(j=i+1;j<n;j++)
                {
                        if(a[i]>=a[j])
                        {
                                t=a[j];
                                a[j]=a[i];
                                a[i]=t;
                        }
                }
        }
        for(i=0;i<n;i++)
        {
                printf("%d\t",a[i]);
        }
        
}
