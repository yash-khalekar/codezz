#include<stdio.h>
int main()
{
        int a[100],i,j,k,n;
        
        printf("enter limit");
        scanf("%d",&n);
        printf("enter %d numbers",n);
        for(i=0;i<n;i++)
        {
                scanf("%d",&a[i]);
        }
        
        for(i=0;i<n;i++)
        {
                for(j=i+1;j<n;j++)
                {
                        if(a[i]==a[j])
                        {
                                for(k=j;k<n;k++)
                                {
                                        a[k]=a[k+1];
                                }
                                n=n-1;
                        }
                }
        }
        printf("new array=\n");
        
        for(i=0;i<n;i++)
        {
                printf("%d\t",a[i]);
        }
}
