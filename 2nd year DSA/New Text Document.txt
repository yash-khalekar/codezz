#include<stdio.h>
int main()
{
        int a[10],i,j,k,n,num,p;
        
        printf("enter limit");
        scanf("%d",&n);
        printf("enter %d numbers",n);
        
        for(i=0;i<n;i++)
        {
                scanf("%d",&a[i]);
        }
        
        printf("enter position from 1 to %d",i);
        scanf("%d",&p);
        printf("Enter the number to insert");
        scanf("%d",&num);
       
        
        for(i=n;i>=p;i--)
        {
             a[i]=a[i-1];
             
        }
        n++;
        a[p-1]=num;
        printf("\n");
        for(i=0;i<n;i++)
        {
                printf("%d ",a[i]);
        }
        
}
